Solved Examples

 Simple calculator by switch

int main() {
double num1, num2; char operation;
cout << "Enter the first number: "; cin >> num1;
cout << "Enter the second number: "; cin >> num2;
cout << "Enter the operation (+, -, *, /): "; cin >> operation;
double result; switch (operation) {
case '+': result = num1 + num2; break;
case '-': result = num1 - num2; break;
case '*': result = num1 * num2; break;
case '/': result = num1 / num2; break;
default: cout << "Invalid operator" << endl;
return 1; }
// Print the result
cout << "Result: " << result << endl;
return 0;}

 The input char is vowel or not by switch (vowel is a,u,i,o,e)

int main() {
// Declare a variable to hold the input character
char ch;
cout << "Enter a character: ";
cin >> ch;
switch (ch) {
case 'a':
case 'e':
case 'i':
case 'o':
case 'u': cout << ch << " is a vowel." << endl; break;
default: cout << ch << " is not a vowel." << endl; break;}
return 0;
}

 Find revers number from given number from keyboard (using while) .

int main() {
int num, rev = 0;
cout << "Enter a number: ";
cin >> num;
while (num > 0) {
rev = rev * 10 + num % 10;
num /= 10;
}
cout << "The reversed number is: " << rev << endl;
return 0;
}
